Zelpware

A voice assistant built on BepInEx. Control your game with natural voice commands, extend with custom plugins, and enable dynamic in-game interactions.

Install v1.0.0 Already have Zelpware?
ExampleMod.cs
using BepInEx;
using Zelpware.Commands.Runtime;

[BepInPlugin("example.zelpware.myplugin", "My Extension", "1.0.0")]
public sealed class MyExtensionPlugin : BaseUnityPlugin {
    private void Awake() {
        // Register your custom command into Zelpware API
        Register(new MyCustomCommand());
    }
}
